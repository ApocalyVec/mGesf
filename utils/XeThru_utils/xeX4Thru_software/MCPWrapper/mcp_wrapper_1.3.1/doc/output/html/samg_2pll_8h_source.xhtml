103 static inline void pll_config_init(struct pll_config *p_cfg,
104  enum pll_source e_src, uint32_t ul_div, uint32_t ul_mul)
105 {
106  uint32_t vco_hz;
107 
108  Assert(e_src < PLL_NR_SOURCES);
109  Assert(ul_div < 2);
110 
111  /* Calculate internal VCO frequency */
112  vco_hz = osc_get_rate(e_src) / ul_div;
113 
114  vco_hz *= ul_mul;
115  Assert(vco_hz >= (PLL_OUTPUT_MIN_HZ - (PLL_OUTPUT_MIN_HZ >> 6)));
116  Assert(vco_hz <= (PLL_OUTPUT_MAX_HZ + (PLL_OUTPUT_MAX_HZ >> 6)));
117 
118  /* PMC hardware will automatically make it mul+1 */
119  p_cfg->ctrl = CKGR_PLLAR_MULA(ul_mul - 1) | CKGR_PLLAR_PLLAEN(ul_div) | CKGR_PLLAR_PLLACOUNT(PLL_COUNT);
120 }
